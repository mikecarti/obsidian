\tau_{i}
\end{bmatrix} = 
\begin{bmatrix}
a_{1} & a_{2} & b_{1} \\
a_{3} & a_{4} & b_{2} \\
c_{1} & c_{2} & 1
\end{bmatrix} \begin{bmatrix}
X \\
Y \\
1
\end{bmatrix}
$$
with transformation $3\times3$ matrix $M$.
